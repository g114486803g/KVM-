转换qcow2
qemu-img convert ttsd-8.39-flat.vmdk -O qcow2 new-vmdk-8-39.qcow2

启动
virt-install -n x2 -r 2048 --vcpus 2 --disk path=/dvd/dobi/mvc/new-vmdk-8-39.qcow2 --network bridge=jbjb --boot hd
